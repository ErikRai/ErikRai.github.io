Sieve of Eratosthenes


def Primes(upperLimit):
primes = list(range(2,upperLimit+1))
for i in primes:
		j = 2
		while i * j <= primes[-1]:
				if i * j in primes:
					primes.remove(i*j)
				j += 1
print('\n')
print('Here are the prime numbers preceeding your entry:')
print('\n')

while True:
	Entry = input('Enter a number: ')  
	x = int(Entry)  
	Primes(x)
